local MarketplaceService = game:GetService("MarketplaceService")
local Players = game:GetService("Players")

local player = Players.LocalPlayer
local shopGui = player.PlayerGui.ShopGui  -- 根據路徑調整

local function playerOwns(gamePassId)
    local success, owns = pcall(function()
        return MarketplaceService:UserOwnsGamePassAsync(player.UserId, gamePassId)
    end)
    return success and owns
end

local function ShopBuy(itemName, amount)
    amount = amount or 1
    print("🛒 正在購買:", itemName, "x" .. amount)
    
    -- Gold+ / 付費道具 強制繞過
    if itemName == "Gold+" or itemName:find("Gold%+") then
        print("✅ 模擬購買 Gold+")
        -- 可自行加入金錢增加邏輯（如果有 Data 遠端）
        return true
    end
    
    -- 角色購買
    if itemName == "Fox Character" then
        workspace.ChangeCharacter:FireServer("FoxCharacter")
        return true
    elseif itemName == "Penguin Character" then
        workspace.ChangeCharacter:FireServer("PenguinCharacter")
        return true
    elseif itemName == "Chicken Character" then
        workspace.ChangeCharacter:FireServer("ChickenCharacter")
        return true
    end
    
    -- 一般道具購買（金幣購買）
    if player.Data and player.Data.Gold then
        player.Data.Gold.Value = player.Data.Gold.Value + 999999  -- 強制金錢足夠
    end
    
    local success = workspace.ItemBoughtFromShop:InvokeServer(itemName, amount)
    if success then
        print("✅ 購買成功:", itemName)
    else
        print("⚠️ 購買失敗，強制標記已擁有")
        player.PlayerGui.ItemGained.LastItemBoughtName.Value = itemName
    end
    return success
end

-- 強制所有按鈕可點擊 + 繞過擁有檢查
for _, btn in ipairs(shopGui:GetDescendants()) do
    if btn:IsA("TextButton") or btn:IsA("ImageButton") then
        if btn:FindFirstChild("Title") or btn:FindFirstChild("TextLabel") then
            local title = btn:FindFirstChild("Title") or btn:FindFirstChild("TextLabel")
            if title and title.Text ~= "" then
                btn.MouseButton1Click:Connect(function()
                    ShopBuy(title.Text)
                end)
            end
        end
    end
end

print("🛍️ 商店 Bypass 已啟用！點擊任何商品即可免費/強制購買")
